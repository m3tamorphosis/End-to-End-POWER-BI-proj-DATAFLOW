Average Loan Amount (High Credit) = 
AVERAGEX(FILTER('Loan_default','Loan_default'[Column credit score bins]="High"),Loan_default[LoanAmount])

Loan by Education Type = 
 COUNTROWS(FILTER('Loan_default',NOT(ISBLANK('Loan_default'[LoanID]))))

Median by Credit Score Bins = 
MEDIANX('Loan_default',Loan_default[LoanAmount])

Total Loan (Credit Bins) = 
    CALCULATE(SUM('Loan_default'[LoanAmount]),'Loan_default'[Age Groups]="Adults",ALLEXCEPT('Loan_default','Loan_default'[Age],'Loan_default'[Age Groups],Loan_default[CreditScore],Loan_default[Column credit score bins]))

Total Loan (Middle age adults) = 
SUMX(FILTER('Loan_default',Loan_default[Age Groups]="Middle Age Adults"),'Loan_default'[LoanAmount])
